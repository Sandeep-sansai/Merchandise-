Admin Components 1.1 AddFlightComponent add-flight.component.ts import { Component } from '@angular/core'; import { FlightService } from '../../services/flight.service'; import { FormBuilder, FormGroup, Validators } from '@angular/forms'; @Component({ selector: 'app-add-flight', templateUrl: './add-flight.component.html', styleUrls: ['./add-flight.component.css'] }) export class AddFlightComponent { addFlightForm: FormGroup; constructor(private fb: FormBuilder, private flightService: FlightService) { this.addFlightForm = this.fb.group({ source: ['', Validators.required], destination: ['', Validators.required], date: ['', Validators.required], time: ['', Validators.required], seats: ['', [Validators.required, Validators.min(1)]], }); } onSubmit() { if (this.addFlightForm.valid) { this.flightService.addFlight(this.addFlightForm.value).subscribe({ next: () => alert('Flight added successfully!'), error: () => alert('Failed to add flight'), }); } } } add-flight.component.html
